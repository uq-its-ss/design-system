# TextInput

<Description of={stories} />

## Installation

```bash
npm i @uqds/react
```

## Usage

```js
import { TextInput } from "@uqds/react";

// Controlled component example.
const [text, setText] = useState('');
return (
  <TextInput
    type="search"
    value={text}
    onChange={setText}
  />
);

```
